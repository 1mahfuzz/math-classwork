\section{Questions}
\Large
1. $x^2+x$ \\
2. $(x-1)^2$ \\
3. $x^2-36$ \\
4. $(a^2-b^2)^2$ \\
5. $a^4 - y^4$ \\
6. $a^2+b^2-c^2-2ab$ \\
7. $1+a^2+a^4$
\newpage
\section{Answers}
1. $x^2+x$ \\
   $=x(x + 1)$\\ \\
2. $(x-1)^2$ \\
   $=(x+1)(x-1)$\\ \\
3. $x^2-36$ \\
   $=x^2-6^2$\\
   $=(x+6)(x-6)$\\ \\
4. $(a^2-b^2)^2$ \\
   $=(a^2-b^2)(a^2-b^2)$\\
   $=(a+b)(a-b)(a+b)(a-b)$\\ \\
5. $a^4 - y^4$ \\
   $=(a^2)^2-(y^2)^2$\\
   $=(a^2+b^2)(a^2-b^2)$\\
   $=(a^2+b^2)(a+b)(a-b)$\\ \\
6. $a^2+b^2-c^2-2ab$ \\
   $=a^2-2ab+b^2-c^2$\\
   $=(a-b)^2-c^2$\\
   $=(a-b+c)(a-b-c)$\\ \\
7. $1+a^2+a^4$\\
   $=(1)^2+2*1*a^2+(a^2)^2-a^2$\\
   $=(1+a^2)^2-(a)^2$\\
   $=(1+a^2+a)(1+a^2-a)$\\
   $=(1+a+a^2)(1-a+a^2)$
